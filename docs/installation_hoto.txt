
-- install pyinstaller
cd git/kedegit

source venv311/bin/activate

pip install pyinstaller

pyinstaller --version

deactivate
--

-- build for Mac folder--

cd git/kedegit

source venv311/bin/activate

pyinstaller --onefile  --name kedegit --distpath ./dist/mac_dist --exclude-module test --paths ./venv311/lib/python3.11/site-packages/ kedehub/__main__.py

./dist/mac_dist/kedegit list-projects

deactivate

cd kedegit

./dist/aws_dist/kedegit list-projects
